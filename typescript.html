Typescript

Install

npm

npm i -g typescript

tsconfig.json

{
    "compilerOptions" : {
        "target" : "esnext",
        "watch" : true,
        "lib" : ["dom", "es2017"]
    }
}

compile

tsc index.tsc

Syntax

implicit value type

let lucky = 23;
lucky = "asd"; // error

explicit value type

let lucky: any = 23;    // if no value assigned: any type
lucky = "asd"; // error

built in types

any
string
number
boolean

make a type

type Style = 'bold' | 'italic' | 23
let font: Style;
let font = 'something'; //error

Interfaces

for objects

interface Person {
    first: string;
    last: string;
    [key: string]: any
}

const person: Person = {
    first: "Zsolt";
    last: "Pap";
    pro: true;
}

Funcions

make function

function pow(x:number, y:number) {
    return Math.pow(x,y);
}
pow(5, 10);

return function

function pow(x:number, y:number): string {
    return Math.pow(x,y).toString();
}
// ? - optional
function hello(name: string, general: boolean?): void {
    if(y == undefined || y == false) {

    } else if (y) {
        console.log(`Hello there, General${name}`);
